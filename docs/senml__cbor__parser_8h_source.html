67  inline unsigned int processMap()
68  {
69  const bool is_indefinite = (peekChar() == (CBOR_MAP | CBOR_VAR_FOLLOWS));
70  uint64_t map_length = 0;
71  size_t read_bytes;
72 
73  if (is_indefinite){
74  log_debug("not supported");
75  }
76  else
77  read_bytes = decode_int(&map_length);
78 
79  unsigned char curState = this->state;
80  size_t i = 0;
81  while (i < map_length) {
82  size_t key_read_bytes, value_read_bytes;
83  this->state = SENML_CBOR_KEY;
84  key_read_bytes = this->parseNext(); /* key */
85  this->state = SENML_CBOR_VALUE;
86  value_read_bytes = this->parseNext(); /* value */
87  if (key_read_bytes == 0 || value_read_bytes == 0) {
88  log_debug("invalid input");
89  break;
90  }
91  read_bytes += key_read_bytes + value_read_bytes;
92  ++i;
93  }
94  this->state = curState; //reset to the original state. was changed inside loop
95  this->ctx.baseValue.baseUint = 0; //if there was a base value, reset it for the next run.
96  return read_bytes;
97  };

99  inline unsigned int processUnsignedInt()
100  {
101  uint64_t val;
102  size_t read_bytes = cbor_deserialize_uint64_t(&val);
103  if(this->state == SENML_CBOR_VALUE){
104  switch (this->curLabel)
105  {
106  case SENML_CBOR_BV_LABEL: this->ctx.baseValue.baseUint = val; break;
107  case SENML_CBOR_V_LABEL:
108  uint64_t calculated = this->ctx.baseValue.baseUint + val;
109  this->setValue((void*)&calculated, sizeof(uint64_t), CBOR_TYPE_UINT);
110  break;
111  }
112  }
113  else if(this->state == SENML_CBOR_KEY) //store the value type (basename, baseunit, value, stringvalue,...)
114  this->curLabel = (int)val;
115  return read_bytes;
116  };

128  this->setValue((void*)&calculated, sizeof(int64_t), CBOR_TYPE_INT);
129  break;
130  }
131  }
132  else if(this->state == SENML_CBOR_KEY) //store the value type (basename, baseunit, value, stringvalue,...)
133  this->curLabel = val;
134  return read_bytes;
